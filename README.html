library(tidyverse)

source('R/vif_func.R')

data(crbs)
data(envdat)

# filter data by depth
# scale variables(subtract mean, divide by sd)
tomod <- envdat %>% 
  filter(depth == 30) %>% 
  left_join(crbs, by = 'CTD') %>% 
  select(-depth) %>% 
  gather('var', 'val', -CTD, -abundances, -pa) %>% 
  group_by(var) %>% 
  mutate(
    val = scale(val)
  ) %>% 
  spread(var, val) 
kpvars <- vif_func(tomod[, -c(1, 2)], trace = F, thresh = 7000)
kpvars

kpvars <- c('Salinity', 'Temperature', 'Spice', 'pH', 'Phosphate')

formin <- kpvars %>% 
  paste(., collapse = ' + ') %>% 
  paste('log(1 + abundances)', ., sep = ' ~ ') %>% 
  formula
gbmod <- gbm(formin, data = tomod, distribution = "gaussian", bag.fraction = 0.9)
# summary(gbmod)

# pa model

  paste(., collapse = ' + ') %>% 
  paste('pa', ., sep = ' ~ ') %>% 
  formula
gbmod <- gbm(formin, data = tomod, distribution = "bernoulli", bag.fraction = 0.9)
